// JavaScript for adding items to the cart
        const cart = {};

        document.querySelectorAll('.add-to-cart').forEach(button => {
            button.addEventListener('click', function() {
                const name = this.getAttribute('data-name');
                const price = parseFloat(this.getAttribute('data-price'));
                if (cart[name]) {
                    cart[name].quantity++;
                } else {
                    cart[name] = { price, quantity: 1 };
                }
                updateQuantity(name, cart[name].quantity);
                alert(`Added "${name}" to the cart. Total: Rs. ${calculateTotal()}`);

                // Send the cart data to the server using a fetch request
                fetch('addtocart.php', {
                    method: 'POST',
                    body: JSON.stringify(cart)
                })
                .then(response => response.json())
                .then(data => {
                    // Handle the server response if needed
                })
                .catch(error => {
                    console.error('Error:', error);
                });
            });
        });

        function calculateTotal() {
            let total = 0;
            for (const item in cart) {
                total += cart[item].price * cart[item].quantity;
            }
            return total.toFixed(2);
        }

        function updateQuantity(name, quantity) {
            const quantityElement = document.querySelector(`.quantity[data-name="${name}"]`);
            if (quantityElement) {
                quantityElement.textContent = quantity;
            }
        }
    
        window.addEventListener("scroll", function() {
            var menuBar = document.getElementById("menu-bar");
            if (window.scrollY > 0) {
                menuBar.classList.add("fixed-menu-bar");
            } else {
                menuBar.classList.remove("fixed-menu-bar");
            }
        });


    </script>
